# Spawning pools are allowed buildings for some of the hiveminded auths.
building_spawning_pool = {
    potential = {
        exists = owner
        owner = { 
            is_hive_empire = yes
            gestalt_controls_spawning = yes
            
            NOT = { has_origin = origin_progenitor_hive }
        }
        or = {
            num_buildings = { type = any value < 15 }
            owner = { is_ai = no }
        }
    }

    destroy_trigger = {
        exists = owner
        owner = { 
            OR = {
                is_hive_empire = no
                gestalt_controls_spawning = no
                has_origin = origin_progenitor_hive
            }
        }
    }
}
